### 普通父传子
:pushpin:
 通过pops传递，接受用行参，也可以结构赋值

```javascript
function Button(props) {
  const handleClick=()=>{
    console.log(props.prop)
  }
  return <button onClick={handleClick}>click</button>
}
function App() {
  
  let prop='ljy'
  return (
    <div className="App">
      <Button prop={prop}/>
      
    </div>
  );
```

### 类似vue中的插槽的方式，通过在子组件标签传入，从父组件的	.children	中获取
```javascript
function Button(props) {
  const handleClick=()=>{
    console.log(props)
  }
  return (
    <div>
       <button onClick={handleClick}>click</button>
       <div>
       {props.children}
       </div>
    </div>
    
  )
}
function App() {
  
  let prop='ljy'
  return (
    <div className="App">
      <Button prop={prop}>
        <div>hello</div>
      </Button>
      
    </div>
  );
}
```



### 子传父
父定义一个函数，并传递给子组件，子组件调用该函数

```jsx
function Button(props) {
  const name='ljuy'
  const age='20'
  const handleClick=()=>{
    console.log(props)
    props.getSonData(name,age,{name:'dyq',desc:'i like you'})
  }
  return (
    <div>
       <button onClick={handleClick}>click</button>
       <div>
       {props.children}
       </div>
    </div>
    
  )
}
function App() {
  
  let prop='ljy'
  const divDom=useRef(null)
  const getSonData=(...rest)=>{
    console.log(rest)
  }
  return (
    <div className="App">
      <Button prop={prop} getSonData={getSonData}>
        <div ref={divDom}>hello</div>
      </Button>
      
    </div>
  );
}
```
